    // Instruction reconnue (Affectation)

    // Instruction qui sera ignorée dans le bloc (If)

} catch (DivisionByZero $ex) { // Teste la structure Catch avec un identifiant de type

    // Un autre bloc Try/Catch imbriqué !

        $log = "ok";
    } catch (Exception $e) {

    }

    // Instruction reconnue (Affectation/Expression)
    $a = $b * 10 - 5;
